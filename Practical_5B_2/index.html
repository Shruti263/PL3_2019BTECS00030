Write a code in any programming language that converts Centigrade to Fahrenheit. 


          #include 
          using namespace std;
          float CentigradetoFahrenheit(float val)
          {
              return ((val * 9.0 / 5.0) + 32.0);
          }
          
          int main()
          {
              float val = 0.0;
              cout << CentigradetoFahrenheit(val);
              return 0;
          }

Output: 32.0
